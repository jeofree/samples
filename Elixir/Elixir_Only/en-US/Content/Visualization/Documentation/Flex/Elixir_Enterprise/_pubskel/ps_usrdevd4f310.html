Creating a graph with nodes and links

The Graph class is a FlexÂ® Group that displays and lays out graphical nodes and links.
The following figure shows how to display a basic graph using the Graph component.
<?xml version="1.0" encoding="utf-8"?>
<s:Application xmlns:fx="http://ns.adobe.com/mxml/2009" 
               xmlns:d="http://www.ibm.com/xmlns/prod/ilog/flex/diagram/2010" 
               xmlns:s="library://ns.adobe.com/flex/spark" 
               minWidth="1024" minHeight="768">
  <s:layout>
    <s:VerticalLayout paddingBottom="20" paddingLeft="20" paddingRight="20" paddingTop="20"/>
  </s:layout>
  <d:Graph automaticGraphLayout="true">
    <d:Node id="node1" label="node 1"/>
    <d:Node id="node2" label="node 2"/>
    <d:Node id="node3" label="node 3"/>
    <d:Node id="node4" label="node 4"/>
    <d:Node id="node5" label="node 5"/>
    <d:Node id="node6" label="node 6"/>
    
    <d:Link startNode="{node1}" endNode="{node3}"/>
    <d:Link startNode="{node1}" endNode="{node2}"/>
    <d:Link startNode="{node1}" endNode="{node4}"/>
    <d:Link startNode="{node4}" endNode="{node5}"/>
    <d:Link startNode="{node4}" endNode="{node6}"/>
    <d:nodeLayout>
      <d:TreeLayout/>
    </d:nodeLayout>
  </d:Graph>
</s:Application>
